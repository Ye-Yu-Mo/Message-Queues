69class StringPiece {
70 private:
71 const char* ptr_;
72 int length_;
73
74 public:
75 // We provide non-explicit singleton constructors so users can pass
76 // in a "const char*" or a "string" wherever a "StringPiece" is
77 // expected.
78 StringPiece()
79 : ptr_(NULL), length_(0) { }
80 StringPiece(const char* str)
81 : ptr_(str), length_(static_cast<int>(strlen(ptr_))) { }
82 StringPiece(const unsigned char* str)
83 : ptr_(reinterpret_cast<const char*>(str)),
84 length_(static_cast<int>(strlen(ptr_))) { }
85 StringPiece(const string& str)
86 : ptr_(str.data()), length_(static_cast<int>(str.size())) { }
87 StringPiece(const char* offset, int len)
88 : ptr_(offset), length_(len) { }
89
90 // data() may return a pointer to a buffer with embedded NULs, and the
91 // returned buffer may or may not be null terminated. Therefore it is
92 // typically a mistake to pass data() to a routine that expects a NUL
93 // terminated string. Use "as_string().c_str()" if you really need to do
94 // this. Or better yet, change your routine so it does not rely on NUL
95 // termination.
96 const char* data() const { return ptr_; }
97 int size() const { return length_; }
98 bool empty() const { return length_ == 0; }
99 const char* begin() const { return ptr_; }
100 const char* end() const { return ptr_ + length_; }
101
102 void clear() { ptr_ = NULL; length_ = 0; }
103 void set(const char* buffer, int len) { ptr_ = buffer; length_ = len; }
104 void set(const char* str) {
105 ptr_ = str;
106 length_ = static_cast<int>(strlen(str));
107 }
108 void set(const void* buffer, int len) {
109 ptr_ = reinterpret_cast<const char*>(buffer);
110 length_ = len;
111 }
112
113 char operator[](int i) const { return ptr_[i]; }
114
115 void remove_prefix(int n) {
116 ptr_ += n;
117 length_ -= n;
118 }
119
120 void remove_suffix(int n) {
121 length_ -= n;
122 }
123
124 bool operator==(const StringPiece& x) const {
125 return ((length_ == x.length_) &&
126 (memcmp(ptr_, x.ptr_, length_) == 0));
127 }
128 bool operator!=(const StringPiece& x) const {
129 return !(*this == x);
130 }
131
132#define STRINGPIECE_BINARY_PREDICATE(cmp,auxcmp) \
133 bool operator cmp (const StringPiece& x) const { \
134 int r = memcmp(ptr_, x.ptr_, length_ < x.length_ ? length_ : x.length_); \
135 return ((r auxcmp 0) || ((r == 0) && (length_ cmp x.length_))); \
136 }
137 STRINGPIECE_BINARY_PREDICATE(<, <);
138 STRINGPIECE_BINARY_PREDICATE(<=, <);
139 STRINGPIECE_BINARY_PREDICATE(>=, >);
140 STRINGPIECE_BINARY_PREDICATE(>, >);
141#undef STRINGPIECE_BINARY_PREDICATE
142
143 int compare(const StringPiece& x) const {
144 int r = memcmp(ptr_, x.ptr_, length_ < x.length_ ? length_ : x.length_);
145 if (r == 0) {
146 if (length_ < x.length_) r = -1;
147 else if (length_ > x.length_) r = +1;
148 }
149 return r;
150 }
151
152 string as_string() const {
153 return string(data(), size());
154 }
155
156 void CopyToString(string* target) const {
157 target->assign(ptr_, length_);
158 }
159
160 // Does "this" start with "x"
161 bool starts_with(const StringPiece& x) const {
162 return ((length_ >= x.length_) && (memcmp(ptr_, x.ptr_, x.length_) == 0));
163 }
164};
