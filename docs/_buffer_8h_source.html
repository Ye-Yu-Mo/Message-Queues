45 static const size_t kCheapPrepend = 8;

48 explicit Buffer(size_t initialSize = kInitialSize)
49 : buffer_(kCheapPrepend + initialSize),
50 readerIndex_(kCheapPrepend),
51 writerIndex_(kCheapPrepend)
52 {
53 assert(readableBytes() == 0);
54 assert(writableBytes() == initialSize);
55 assert(prependableBytes() == kCheapPrepend);
56 }

61 void swap(Buffer& rhs)
62 {
63 buffer_.swap(rhs.buffer_);

65 std::swap(writerIndex_, rhs.writerIndex_);
66 }

114 {
115 assert(len <= readableBytes());
116 if (len < readableBytes())
117 {
118 readerIndex_ += len;

354 void prepend(const void* /*restrict*/ data, size_t len)
355 {
356 assert(len <= prependableBytes());
357 readerIndex_ -= len;
358 const char* d = static_cast<const char*>(data);
359 std::copy(d, d+len, begin()+readerIndex_);
360 }

362 void shrink(size_t reserve)
363 {
364 // FIXME: use vector::shrink_to_fit() in C++ 11 if possible.
365 Buffer other;
366 other.ensureWritableBytes(readableBytes()+reserve);
367 other.append(toStringPiece());
368 swap(other);
369 }

390 void makeSpace(size_t len)
391 {
392 if (writableBytes() + prependableBytes() < len + kCheapPrepend)
393 {
394 // FIXME: move readable data
395 buffer_.resize(writerIndex_+len);
396 }
397 else
398 {
399 // move readable data to the front, make space inside buffer
400 assert(kCheapPrepend < readerIndex_);
401 size_t readable = readableBytes();
402 std::copy(begin()+readerIndex_,
403 begin()+writerIndex_,
404 begin()+kCheapPrepend);
405 readerIndex_ = kCheapPrepend;
406 writerIndex_ = readerIndex_ + readable;
407 assert(readable == readableBytes());
408 }
409 }

411 private:

413 size_t readerIndex_;
414 size_t writerIndex_;
