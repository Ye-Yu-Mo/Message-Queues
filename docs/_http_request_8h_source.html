53 bool setMethod(const char* start, const char* end)
54 {
55 assert(method_ == kInvalid);
56 string m(start, end);
57 if (m == "GET")

59 method_ = kGet;

135 void addHeader(const char* start, const char* colon, const char* end)
136 {
137 string field(start, colon);
138 ++colon;
139 while (colon < end && isspace(*colon))
140 {
141 ++colon;
142 }
143 string value(colon, end);
144 while (!value.empty() && isspace(value[value.size()-1]))
145 {
146 value.resize(value.size()-1);
147 }
148 headers_[field] = value;
149 }
150
151 string getHeader(const string& field) const
152 {
153 string result;
154 std::map<string, string>::const_iterator it = headers_.find(field);
155 if (it != headers_.end())
156 {
157 result = it->second;
158 }
159 return result;
160 }
161
162 const std::map<string, string>& headers() const
163 { return headers_; }

167 std::swap(method_, that.method_);
168 std::swap(version_, that.version_);
169 path_.swap(that.path_);
170 query_.swap(that.query_);
171 receiveTime_.swap(that.receiveTime_);
172 headers_.swap(that.headers_);
173 }
174
175 private:
