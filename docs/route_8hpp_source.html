98 static bool route(ExchangeType type, const std::string &routing_key, const std::string &binding_key)
99 {
100 if (type == ExchangeType::DIRECT)
101 return routing_key == binding_key;
102 if (type == ExchangeType::FANOUT)
103 return true;
104
105 // 动态规划进行匹配
106 // 二维数组标记每次匹配的结果 通过最终数组末尾的位置来判定整体是否匹配成功
107 // 初始化 dp[0][0] = 1;
108 // 遇到 普通字符
109 // 如果匹配成功 dp[i][j] = dp[i-1][j-1];
110 // 如果匹配失败 dp[i][j] = 0;
111 // 遇到 通配符 `#`
112 // 初始化 bindingkey以 `#`起始时 对应行第一个字符初始化为1
113 // dp[i][j] = dp[i][j-1] || dp[i-1][j] || dp[i-1][j-1];
114 // 遇到 通配符 `*` 始终认为单词匹配成功
115 // dp[i][j] = dp[i-1][j-1];
116 std::vector<std::string> bkeys, rkeys;
117 // 字符串风格 得到单词数组
118 int n_bkey = StrHelper::split(binding_key, ".", bkeys);
119 int n_rkey = StrHelper::split(routing_key, ".", rkeys);
120 std::vector<std::vector<bool>> dp(n_bkey + 1, std::vector<bool>(n_rkey + 1, false));
121 // 初始化
122 dp[0][0] = true;
123 // bindingkey以#起始的行第0列置为1
124 for (int i = 1; i <= bkeys.size(); i++)
125 {
126 if (bkeys[i - 1] == "#")
127 {
128 dp[i][0] = true;
129 continue;
130 }
131 break;
132 }
133 // 匹配
134 for (int i = 1; i <= n_bkey; i++)
135 {
136 for (int j = 1; j <= n_rkey; j++)
137 {
138 if (bkeys[i - 1] == rkeys[j - 1] || bkeys[i - 1] == "*")
139 {
140 dp[i][j] = dp[i - 1][j - 1];
141 }
142 else if (bkeys[i - 1] == "#")
143 {
144 dp[i][j] = dp[i - 1][j - 1] | dp[i - 1][j] | dp[i][j - 1];
145 }
146 }
147 }
148 return dp[n_bkey][n_rkey];
149 }
