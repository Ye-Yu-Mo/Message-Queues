119 void basicPublish(const basicPublishRequestPtr &req)
120 {
121 // 获取交换机
122 Exchange::ptr ep = _host->selectExchange(req->exchange_name());
123 if (ep.get() == nullptr)
124 basicRespFunc(false, req->rid(), req->cid());
125 // 获取指定交换机的绑定信息
126 MsgQueueBindingMap mqbm = _host->exchangeBindings(req->exchange_name());
127 BasicProperties *properties = nullptr;
128 std::string routing_key;
129
130 if (req->has_properties())
131 {
132 properties = req->mutable_properties();
133 routing_key = properties->routing_key();
134 }
135 for (auto &binding : mqbm)
136 {
137 // 交换路由 找到对应的队列
138 if (Router::route(ep->type, routing_key, binding.second->binding_key))
139 {
140 // 将消息添加到队列中
141 _host->basicPublish(binding.first, properties, req->body());
142 // 向线程池中添加一个消息消费任务(向指定队列的订阅者推送消息)
143 auto task = std::bind(&Channel::consume, this, binding.first);
144 _pool->push(task);
145 }
146 basicRespFunc(true, req->rid(), req->cid());
147 }

191 void consume(const std::string &qname)
192 {
193 MessagePtr mp = _host->basicConsume(qname);
194 if (mp.get() == nullptr)
195 {
196 error(logger, "消费任务失败, 指定队列中没有消息: %s", qname.c_str());
197 return;
198 }
199 Consumer::ptr cp = _cmp->choose(qname);
200 if (cp.get() == nullptr)
201 {
202 error(logger, "消费任务失败, 指定队列中没有消费者: %s", qname.c_str());
203 return;
204 }
205 cp->callback(cp->tag, mp->mutable_payload()->mutable_properties(), mp->payload().body());
206 if (cp->auto_ack == true)
207 _host->basicAck(qname, mp->payload().properties().id());
208 }

222 resp.mutable_properties()->set_delivery_mode(bp->delivery_mode());
223 resp.mutable_properties()->set_routing_key(bp->routing_key());
224 }
