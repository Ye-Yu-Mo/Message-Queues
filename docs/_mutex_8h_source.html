110// Use as data member of a class, eg.
111//
112// class Foo
113// {
114// public:
115// int size() const;
116//
117// private:
118// mutable MutexLock mutex_;
119// std::vector<int> data_ GUARDED_BY(mutex_);
120// };
121class CAPABILITY("mutex") MutexLock : noncopyable
122{
123 public:
124 MutexLock()
125 : holder_(0)
126 {
127 MCHECK(pthread_mutex_init(&mutex_, NULL));
128 }
129
130 ~MutexLock()
131 {
132 assert(holder_ == 0);
133 MCHECK(pthread_mutex_destroy(&mutex_));
134 }
135
136 // must be called when locked, i.e. for assertion
137 bool isLockedByThisThread() const
138 {
139 return holder_ == CurrentThread::tid();
140 }
141
142 void assertLocked() const ASSERT_CAPABILITY(this)
143 {
144 assert(isLockedByThisThread());
145 }
146
147 // internal usage
148
149 void lock() ACQUIRE()
150 {
151 MCHECK(pthread_mutex_lock(&mutex_));
152 assignHolder();
153 }
154
155 void unlock() RELEASE()
156 {
157 unassignHolder();
158 MCHECK(pthread_mutex_unlock(&mutex_));
159 }
160
161 pthread_mutex_t* getPthreadMutex() /* non-const */
162 {
163 return &mutex_;
164 }
165
166 private:
167 friend class Condition;
168
169 class UnassignGuard : noncopyable
170 {
171 public:
172 explicit UnassignGuard(MutexLock& owner)
173 : owner_(owner)
174 {
175 owner_.unassignHolder();
176 }
177
178 ~UnassignGuard()
179 {
180 owner_.assignHolder();
181 }
182
183 private:
184 MutexLock& owner_;
185 };
186
187 void unassignHolder()
188 {
189 holder_ = 0;
190 }
191
192 void assignHolder()
193 {
194 holder_ = CurrentThread::tid();
195 }
196
197 pthread_mutex_t mutex_;
198 pid_t holder_;
199};
200
201// Use as a stack variable, eg.
202// int Foo::size() const
203// {
204// MutexLockGuard lock(mutex_);
205// return data_.size();
206// }
207class SCOPED_CAPABILITY MutexLockGuard : noncopyable
208{
209 public:
210 explicit MutexLockGuard(MutexLock& mutex) ACQUIRE(mutex)
211 : mutex_(mutex)
212 {
213 mutex_.lock();
214 }
215
216 ~MutexLockGuard() RELEASE()
217 {
218 mutex_.unlock();
219 }
220
221 private:
222
223 MutexLock& mutex_;
224};
225
226} // namespace muduo
