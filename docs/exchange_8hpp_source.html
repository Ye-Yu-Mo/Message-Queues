46 void setArgs(const std::string &str_args)
47 {
48 std::vector<std::string> sub_args;
49 StrHelper::split(str_args, "&", sub_args);
50 for (auto &str : sub_args)
51 {
52 size_t pos = str.find('=');

55 args.insert(std::make_pair(key, value));

63 std::string getArgs()

198 void declareExchange(const std::string &name,
199 ExchangeType type,
200 bool durable,
201 bool auto_delete,
202 std::unordered_map<std::string, std::string> &args)
203 {
204 std::unique_lock<std::mutex> lock(_mutex);
205 auto it = _exchanges.find(name);
206 if (it != _exchanges.end())
207 return;
208 auto exp = std::make_shared<Exchange>(name, type, durable, auto_delete, args);
209 if (durable)
210 _mapper.insert(exp);
211 _exchanges.insert(std::make_pair(name, exp));
212 }
215 void deleteExchange(const std::string &name)
216 {
217 std::unique_lock<std::mutex> lock(_mutex);
218 auto it = _exchanges.find(name);
219 if (it == _exchanges.end())
220 return;
221 _exchanges.erase(name);
222 if (it->second->durable == true)
223 _mapper.remove(name);
224 }
228 Exchange::ptr selectExchange(const std::string &name)
229 {
230 std::unique_lock<std::mutex> lock(_mutex);
231 auto it = _exchanges.find(name);
232 if (it == _exchanges.end())
233 return Exchange::ptr();
234 return it->second;
235 }
239 bool exists(const std::string &name)
240 {
241 std::unique_lock<std::mutex> lock(_mutex);
242 auto it = _exchanges.find(name);
243 if (it == _exchanges.end())
244 return false;
245 return true;
246 }
248 void clear()
249 {
250 std::unique_lock<std::mutex> lock(_mutex);
251 _mapper.removeTable();
252 _exchanges.clear();
253 }
256 size_t size()
257 {
258 std::unique_lock<std::mutex> lock(_mutex);
259 return _exchanges.size();
260 }
