40// wire format
41//
42// Field Length Content
43//
44// size 4-byte M+N+4
45// tag M-byte could be "RPC0", etc.
46// payload N-byte
47// checksum 4-byte adler32 of tag+payload
48//
49// This is an internal class, you should use ProtobufCodecT instead.
50class ProtobufCodecLite : noncopyable
51{
52 public:
53 const static int kHeaderLen = sizeof(int32_t);
54 const static int kChecksumLen = sizeof(int32_t);
55 const static int kMaxMessageLen = 64*1024*1024; // same as codec_stream.h kDefaultTotalBytesLimit
56
57 enum ErrorCode
58 {
59 kNoError = 0,
60 kInvalidLength,
61 kCheckSumError,
62 kInvalidNameLen,
63 kUnknownMessageType,
64 kParseError,
65 };
66
67 // return false to stop parsing protobuf message
68 typedef std::function<bool (const TcpConnectionPtr&,
69 StringPiece,
70 Timestamp)> RawMessageCallback;
71
72 typedef std::function<void (const TcpConnectionPtr&,
73 const MessagePtr&,
74 Timestamp)> ProtobufMessageCallback;
75
76 typedef std::function<void (const TcpConnectionPtr&,
77 Buffer*,
78 Timestamp,
79 ErrorCode)> ErrorCallback;
80
81 ProtobufCodecLite(const ::google::protobuf::Message* prototype,
82 StringPiece tagArg,
83 const ProtobufMessageCallback& messageCb,
84 const RawMessageCallback& rawCb = RawMessageCallback(),
85 const ErrorCallback& errorCb = defaultErrorCallback)
86 : prototype_(prototype),
87 tag_(tagArg.as_string()),
88 messageCallback_(messageCb),
89 rawCb_(rawCb),
90 errorCallback_(errorCb),
91 kMinMessageLen(tagArg.size() + kChecksumLen)
92 {
93 }

95 virtual ~ProtobufCodecLite() = default;
96
97 const string& tag() const { return tag_; }
98
99 void send(const TcpConnectionPtr& conn,
100 const ::google::protobuf::Message& message);
101
102 void onMessage(const TcpConnectionPtr& conn,
103 Buffer* buf,
104 Timestamp receiveTime);
105
106 virtual bool parseFromBuffer(StringPiece buf, google::protobuf::Message* message);
107 virtual int serializeToBuffer(const google::protobuf::Message& message, Buffer* buf);
108
109 static const string& errorCodeToString(ErrorCode errorCode);
110
111 // public for unit tests
112 ErrorCode parse(const char* buf, int len, ::google::protobuf::Message* message);
113 void fillEmptyBuffer(muduo::net::Buffer* buf, const google::protobuf::Message& message);
114
115 static int32_t checksum(const void* buf, int len);
116 static bool validateChecksum(const char* buf, int len);
117 static int32_t asInt32(const char* buf);
118 static void defaultErrorCallback(const TcpConnectionPtr&,
119 Buffer*,
120 Timestamp,
121 ErrorCode);
122
123 private:
124 const ::google::protobuf::Message* prototype_;
125 const string tag_;
126 ProtobufMessageCallback messageCallback_;
127 RawMessageCallback rawCb_;
128 ErrorCallback errorCallback_;
129 const int kMinMessageLen;
130};

132template<typename MSG, const char* TAG, typename CODEC=ProtobufCodecLite> // TAG must be a variable with external linkage, not a string literal
133class ProtobufCodecLiteT
134{
135 static_assert(std::is_base_of<ProtobufCodecLite, CODEC>::value, "CODEC should be derived from ProtobufCodecLite");
136 public:
137 typedef std::shared_ptr<MSG> ConcreteMessagePtr;
138 typedef std::function<void (const TcpConnectionPtr&,
139 const ConcreteMessagePtr&,
140 Timestamp)> ProtobufMessageCallback;
141 typedef ProtobufCodecLite::RawMessageCallback RawMessageCallback;
142 typedef ProtobufCodecLite::ErrorCallback ErrorCallback;
143
144 explicit ProtobufCodecLiteT(const ProtobufMessageCallback& messageCb,
145 const RawMessageCallback& rawCb = RawMessageCallback(),
146 const ErrorCallback& errorCb = ProtobufCodecLite::defaultErrorCallback)
147 : messageCallback_(messageCb),
148 codec_(&MSG::default_instance(),
149 TAG,
150 std::bind(&ProtobufCodecLiteT::onRpcMessage, this, _1, _2, _3),
151 rawCb,
152 errorCb)
153 {
154 }
155
156 const string& tag() const { return codec_.tag(); }
157
158 void send(const TcpConnectionPtr& conn,
159 const MSG& message)
160 {
161 codec_.send(conn, message);
162 }
163
164 void onMessage(const TcpConnectionPtr& conn,
165 Buffer* buf,
166 Timestamp receiveTime)
167 {
168 codec_.onMessage(conn, buf, receiveTime);
169 }
170
171 // internal
172 void onRpcMessage(const TcpConnectionPtr& conn,
173 const MessagePtr& message,
174 Timestamp receiveTime)
175 {
176 messageCallback_(conn, ::muduo::down_pointer_cast<MSG>(message), receiveTime);
177 }
178
179 void fillEmptyBuffer(muduo::net::Buffer* buf, const MSG& message)
180 {
181 codec_.fillEmptyBuffer(buf, message);
182 }
183
184 private:
185 ProtobufMessageCallback messageCallback_;
186 CODEC codec_;
187};
